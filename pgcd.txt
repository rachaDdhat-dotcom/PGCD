#include <stdio.h>

int main() {
    int a, b, x, y, r, etape = 1;

    // Lire les deux nombres
    printf("Entrez le premier nombre : ");
    scanf("%d", &a);
    printf("Entrez le deuxième nombre : ");
    scanf("%d", &b);

    // Sauvegarder les valeurs initiales
    x = a;
    y = b;

    printf("\nCalcul du PGCD(%d, %d)\n\n", x, y);

    // Algorithme d’Euclide avec affichage des étapes
    while (b != 0) {
        r = a % b;
        printf("Étape %d : %d = %d * (%d) + %d\n", etape, a, b, a / b, r);
        a = b;
        b = r;
        etape++;
    }

    // Résultat final
    printf("\nLe PGCD de %d et %d est : %d\n", x, y, a);

    return 0;
}
